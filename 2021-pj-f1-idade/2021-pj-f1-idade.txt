```cpp
#include <iostream>

int main() {
    int age1, age2, age3;

    // Ordenar as idades em ordem crescente
    int minAge = std::min({age1, age2, age3});

    return 0;
}
```

* Finalmente, imprimimos a idade de Camila com `std::cout`.
